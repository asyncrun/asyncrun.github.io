# 二分查找

```cpp
#pragma once
#include <iostream>
using namespace std;

int binarySearchRecursive(int arr[], int l, int r, int x)
{
	if(l <= r)
	{
		int mid = l + (r - l) / 2;
		if (arr[mid] == x)
		{
			return mid;
		}

		if(arr[mid] > x)
		{
			return binarySearchRecursive(arr, l, mid - 1, x);
		}

		return binarySearchRecursive(arr, mid + 1, r, x);
	}

	return -1;
}

int binarySearchIterative(int arr[], int l, int r, int x)
{
	while (l <= r)
	{
		int mid = l + (r - l) / 2;
		if (arr[mid] == x) return mid;
		if (arr[mid] < x) return l = mid + 1;
		if (arr[mid] > x) return r = mid - 1;
	}

	return  -1;
}

void test_binarySearchRecursive()
{
	int arr[] = {2, 3, 4, 10, 50};
	int x = 10;
	int n = sizeof(arr) / sizeof(arr[0]);
	int result = binarySearchRecursive(arr, 0, n - 1, x);
	result == -1 ? cout << "Element is not present in array"
		: cout << "Element is present at index " << result << endl;
}



void test_binarySearchIterative()
{
	int arr[] = { 2, 3, 4, 10, 50 };
	int x = 3;
	int n = sizeof(arr) / sizeof(arr[0]);
	int result = binarySearchIterative(arr, 0, n - 1, x);
	result == -1 ? cout << "Element is not present in array"
		: cout << "Element is present at index " << result << endl;
}
```
